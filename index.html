            // --- Affinity Analysis Logic ---

            const totalBaskets = 1000;

            const milkSlider = document.getElementById('baskets-with-milk');

            const breadSlider = document.getElementById('baskets-with-bread');

            const bothSlider = document.getElementById('baskets-with-both');

            const milkValue = document.getElementById('baskets-with-milk-value');

            const breadValue = document.getElementById('baskets-with-bread-value');

            const bothValue = document.getElementById('baskets-with-both-value');

            

            const supportMetricEl = document.getElementById('support-metric');

            const confidenceMetricEl = document.getElementById('confidence-metric');

            const liftMetricEl = document.getElementById('lift-metric');


            function calculateAffinityMetrics() {

                const numMilk = parseInt(milkSlider.value);

                const numBread = parseInt(breadSlider.value);

                const numBoth = parseInt(bothSlider.value);


                milkValue.textContent = numMilk;

                breadValue.textContent = numBread;

                bothValue.textContent = numBoth;

                bothSlider.max = Math.min(numMilk, numBread); // Ensure 'both' can't be more than either individual item


                const support = (numBoth / totalBaskets) * 100;

                const confidence = numMilk > 0 ? (numBoth / numMilk) * 100 : 0;

                const lift = (numMilk > 0 && numBread > 0) ? (support / 100) / ((numMilk / totalBaskets) * (numBread / totalBaskets)) : 0;

                

                supportMetricEl.textContent = `${support.toFixed(1)}%`;

                confidenceMetricEl.textContent = `${confidence.toFixed(1)}%`;

                liftMetricEl.textContent = lift.toFixed(2);

                

                if (lift > 1) {

                    liftMetricEl.classList.remove('text-red-600');

                    liftMetricEl.classList.add('text-green-600');

                } else {

                    liftMetricEl.classList.remove('text-green-600');

                    liftMetricEl.classList.add('text-red-600');

                }

            }


            [milkSlider, breadSlider, bothSlider].forEach(slider => {

                slider.addEventListener('input', calculateAffinityMetrics);

            });


            // --- Initial Load ---

            const init = () => {

                createAnomalyChart();

                calculateAffinityMetrics();

                document.getElementById('nav-anomaly').click();

            };


            init();

        });
